    triggers {

    stages {
        stage('Build') {
            steps {

                echo 'Building...'
				

				
				if($found){
				echo 'File Exists'
				}
				
            }
        }
		
		stage('Test') {
            steps {

                echo 'Testing...'


            }
        }
		
		stage('Deploy') {
            steps {

                echo 'Deploying...'

				

			}
        }
    }
 
    post{
        success{

                echo '  SUCCESS!'

        }
    }
}
